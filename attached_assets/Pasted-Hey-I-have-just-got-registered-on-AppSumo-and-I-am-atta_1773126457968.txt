const crypto = require('crypto');
function validateWebhook(body, timestamp, signature, apiKey) {
  // Combine timestamp + body
  const message = timestamp + JSON.stringify(body);
  
  // Generate signature using your API key
  const calculatedSignature = crypto
    .createHmac('sha256', apiKey)
    .update(message)
    .digest('hex');
  
  // Compare signatures
  return signature === calculatedSignature;
}

function requireFeature(featureName) {
  return async (req, res, next) => {
    const user = req.user;
    const tierConfig = APPSUMO_TIERS[user.appsumo_tier];
    
    if (tierConfig && tierConfig.features.includes(featureName)) {
      next();
    } else {
      res.status(403).json({
        error: 'Upgrade required',
        message: `This feature requires ${featureName}`
      });
    }
  };
}

4.1 License Key Generation Logic
// Backend: /api/partner/licenses/generate
async function generateLicenseKeys(partnerId, tier, quantity, batchName, generatedBy) {
  // 1. Create batch record
  const batchId = `${partner.key_prefix}-${tier}-${Date.now()}`;
  const batch = await db.key_generation_batches.create({
    batch_id: batchId,
    partner_id: partnerId,
    tier: tier,
    quantity: quantity,
    generated_by_user_id: generatedBy.userId,
    generated_by_type: generatedBy.type,  // 'admin' or 'partner'
    notes: batchName
  });
  // 2. Generate UUID keys
  const keys = [];
  for (let i = 0; i < quantity; i++) {
    const licenseKey = uuidv4();  // e.g., a1b2c3d4-1234-5678-9abc-def012345678
    
    keys.push({
      license_key: licenseKey,
      partner_id: partnerId,
      tier: tier,
      status: 'generated',
      generation_batch_id: batchId,
      generated_at: new Date()
    });
  }
  // 3. Bulk insert into database
  await db.partner_license_keys.bulkCreate(keys);
  // 4. Log events
  for (const key of keys) {
    await db.partner_license_events.create({
      license_key: key.license_key,
      partner_id: partnerId,
      event_type: 'generated',
      new_status: 'generated',
      triggered_by: 'manual',
      tier: tier
    });
  }
  return {
    batchId,
    keysGenerated: quantity,
    keys: keys.map(k => k.license_key)
  };
}

// Backend: /api/partner/licenses/export
async function exportLicenseKeys(partnerId, filters) {
  // Build query based on filters
  const query = {
    partner_id: partnerId,
    ...filters  // status, tier, batchId, dateRange, etc.
  };
  const keys = await db.partner_license_keys.findAll({
    where: query,
    order: [['generated_at', 'DESC']]
  });
  // Format as CSV
  const csv = [
    ['License Key', 'Tier', 'Status', 'Generated At', 'Consumed At', 'Redeemed At', 'Batch ID'].join(','),
    ...keys.map(k => [
      k.license_key,
      k.tier,
      k.status,
      k.generated_at,
      k.consumed_at || '',
      k.redeemed_at || '',
      k.generation_batch_id
    ].join(','))
  ].join('\n');
  return csv;
}

4.3 Webhook Handler (Updated with Partner Context)
// Backend: /api/webhooks/partner
async function handlePartnerWebhook(req, res) {
  const { license_key, event, tier, prev_license_key, ...payload } = req.body;
  
  // 1. Validate webhook signature (HMAC)
  const signature = req.headers['x-appsumo-signature'];
  const timestamp = req.headers['x-appsumo-timestamp'];
  
  // Find partner by license_key
  const licenseRecord = await db.partner_license_keys.findOne({
    where: { license_key },
    include: [{ model: db.partners }]
  });
  
  if (!licenseRecord) {
    return res.status(404).json({ success: false, error: 'License key not found' });
  }
  
  const partner = licenseRecord.partner;
  
  // Validate signature using partner's webhook_secret
  if (!validateHMACSignature(req.body, timestamp, signature, partner.webhook_secret)) {
    return res.status(401).json({ success: false, error: 'Invalid signature' });
  }
  
  // 2. Handle test webhooks
  if (payload.test === true) {
    return res.json({ event, success: true });
  }
  
  // 3. Process event
  switch (event) {
    case 'purchase':
      await handlePurchaseEvent(licenseRecord, payload);
      break;
    case 'activate':
      await handleActivateEvent(licenseRecord, payload);
      break;
    case 'upgrade':
      await handleUpgradeEvent(licenseRecord, prev_license_key, tier, payload);
      break;
    case 'downgrade':
      await handleDowngradeEvent(licenseRecord, prev_license_key, tier, payload);
      break;
    case 'deactivate':
      await handleDeactivateEvent(licenseRecord, payload);
      break;
  }
  
  // 4. Return success
  return res.json({ event, success: true });
}
async function handlePurchaseEvent(licenseRecord, payload) {
  await db.partner_license_keys.update({
    status: 'consumed',
    consumed_at: new Date()
  }, {
    where: { license_key: licenseRecord.license_key }
  });
  
  await db.partner_license_events.create({
    license_key: licenseRecord.license_key,
    partner_id: licenseRecord.partner_id,
    event_type: 'consumed',
    previous_status: 'generated',
    new_status: 'consumed',
    triggered_by: 'webhook',
    webhook_payload: payload,
    tier: licenseRecord.tier
  });
}
async function handleActivateEvent(licenseRecord, payload) {
  await db.partner_license_keys.update({
    status: 'redeemed',
    redeemed_at: new Date()
  }, {
    where: { license_key: licenseRecord.license_key }
  });
  
  await db.partner_license_events.create({
    license_key: licenseRecord.license_key,
    partner_id: licenseRecord.partner_id,
    event_type: 'redeemed',
    previous_status: 'consumed',
    new_status: 'redeemed',
    triggered_by: 'webhook',
    webhook_payload: payload,
    tier: licenseRecord.tier
  });
}
async function handleUpgradeEvent(licenseRecord, prevLicenseKey, newTier, payload) {
  // Mark old key as upgraded
  const oldKey = await db.partner_license_keys.findOne({
    where: { license_key: prevLicenseKey }
  });
  
  await db.partner_license_keys.update({
    status: 'upgraded',
    upgraded_at: new Date(),
    upgraded_to_key: licenseRecord.license_key
  }, {
    where: { license_key: prevLicenseKey }
  });
  
  // Mark new key as consumed and redeemed immediately
  await db.partner_license_keys.update({
    status: 'redeemed',
    consumed_at: new Date(),
    redeemed_at: new Date(),
    previous_key: prevLicenseKey,
    user_id: oldKey.user_id  // Transfer user
  }, {
    where: { license_key: licenseRecord.license_key }
  });
  
  // Update user
  await db.users.update({
    partner_license_key: licenseRecord.license_key,
    partner_tier: newTier
  }, {
    where: { id: oldKey.user_id }
  });
  
  // Log events
  await db.partner_license_events.create({
    license_key: licenseRecord.license_key,
    partner_id: licenseRecord.partner_id,
    event_type: 'upgraded',
    previous_status: 'generated',
    new_status: 'redeemed',
    triggered_by: 'webhook',
    webhook_payload: payload,
    previous_tier: oldKey.tier,
    tier: newTier,
    user_id: oldKey.user_id
  });
}

// Backend: /api/auth/partner/callback
async function handlePartnerOAuthCallback(req, res) {
  const { code } = req.query;
  
  if (!code) {
    return res.status(400).send('Missing authorization code');
  }
  
  // Determine which partner based on request source
  // (Could be from subdomain, query param, or session)
  const partnerName = req.query.partner || req.session.partner || 'appsumo';
  const partner = await db.partners.findOne({ where: { name: partnerName } });
  
  try {
    // 1. Exchange code for access token
    const tokenResponse = await fetch(`${partner.oauth_redirect_url}/token/`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        client_id: partner.oauth_client_id,
        client_secret: partner.oauth_client_secret,
        code: code,
        redirect_uri: `${process.env.APP_URL}/api/auth/partner/callback?partner=${partnerName}`,
        grant_type: 'authorization_code'
      })
    });
    
    const { access_token, refresh_token } = await tokenResponse.json();
    
    // 2. Fetch license details
    const licenseResponse = await fetch(`${partner.oauth_redirect_url}/license_key/?access_token=${access_token}`);
    const { license_key, status } = await licenseResponse.json();
    
    // 3. Find license in our database
    const licenseRecord = await db.partner_license_keys.findOne({
      where: { license_key, partner_id: partner.id }
    });
    
    if (!licenseRecord) {
      return res.status(404).send('License not found');
    }
    
    // 4. Check if already linked to user
    if (licenseRecord.user_id) {
      const existingUser = await db.users.findByPk(licenseRecord.user_id);
      
      // Auto-login
      req.session.userId = existingUser.id;
      return res.redirect('/dashboard');
    }
    
    // 5. New user - store license data in session
    req.session.pendingPartnerLicense = {
      license_key,
      partner_id: partner.id,
      partner_name: partner.name,
      tier: licenseRecord.tier,
      access_token,
      refresh_token
    };
    
    // 6. Redirect to onboarding
    res.redirect(`/onboarding?source=partner&partner=${partnerName}`);
    
  } catch (error) {
    console.error('OAuth error:', error);
    res.status(500).send('Authentication failed');
  }
}

5.1 Real-time Dashboard Stats
async function getPartnerDashboardStats(partnerId) {
  const stats = await db.partner_license_keys.findAll({
    where: { partner_id: partnerId },
    attributes: [
      'status',
      'tier',
      [db.fn('COUNT', db.col('id')), 'count']
    ],
    group: ['status', 'tier']
  });
  
  // Transform into dashboard format
  const dashboard = {
    total_generated: 0,
    by_status: {
      generated: 0,
      consumed: 0,
      redeemed: 0,
      upgraded: 0,
      downgraded: 0,
      deactivated: 0
    },
    by_tier: {
      1: { generated: 0, consumed: 0, redeemed: 0, available: 0 },
      2: { generated: 0, consumed: 0, redeemed: 0, available: 0 },
      3: { generated: 0, consumed: 0, redeemed: 0, available: 0 }
    }
  };
  
  stats.forEach(stat => {
    const count = parseInt(stat.get('count'));
    dashboard.total_generated += count;
    dashboard.by_status[stat.status] += count;
    
    if (stat.tier) {
      dashboard.by_tier[stat.tier][stat.status] = count;
      if (stat.status === 'generated') {
        dashboard.by_tier[stat.tier].available += count;
      }
    }
  });
  
  return dashboard;
}
5.2 Conversion Funnel Report
async function getConversionFunnel(partnerId, dateRange) {
  const funnel = await db.sequelize.query(`
    SELECT 
      COUNT(*) FILTER (WHERE status IN ('generated', 'consumed', 'redeemed')) as total_generated,
      COUNT(*) FILTER (WHERE status IN ('consumed', 'redeemed')) as total_consumed,
      COUNT(*) FILTER (WHERE status = 'redeemed') as total_redeemed
    FROM partner_license_keys
    WHERE partner_id = :partnerId
      AND generated_at BETWEEN :startDate AND :endDate
  `, {
    replacements: { partnerId, ...dateRange },
    type: db.QueryTypes.SELECT
  });
  
  const data = funnel[0];
  
  return {
    generated: data.total_generated,
    consumed: data.total_consumed,
    redeemed: data.total_redeemed,
    consumed_rate: (data.total_consumed / data.total_generated * 100).toFixed(2) + '%',
    redemption_rate: (data.total_redeemed / data.total_consumed * 100).toFixed(2) + '%'
  };
}
5.3 Daily Activity Report
async function getDailyActivity(partnerId, days = 30) {
  const startDate = new Date();
  startDate.setDate(startDate.getDate() - days);
  
  const activity = await db.partner_license_events.findAll({
    where: {
      partner_id: partnerId,
      created_at: { [db.Op.gte]: startDate }
    },
    attributes: [
      [db.fn('DATE', db.col('created_at')), 'date'],
      'event_type',
      [db.fn('COUNT', db.col('id')), 'count']
    ],
    group: ['date', 'event_type'],
    order: [['date', 'ASC']]
  });
  
  // Format for charting
  const dailyData = {};
  activity.forEach(row => {
    const date = row.get('date');
    if (!dailyData[date]) {
      dailyData[date] = { consumed: 0, redeemed: 0, upgraded: 0, deactivated: 0 };
    }
    dailyData[date][row.event_type] = parseInt(row.get('count'));
  });
  
  return dailyData;
}

    userId: partnerUser.id,
    partnerId: partnerUser.partner_id,
    partnerName: partnerUser.partner.name,
    role: partnerUser.role,
    token: generateJWT(partnerUser)
  };
}

// Middleware to protect partner routes
function requirePartnerAuth(req, res, next) {
  if (!req.session.partnerUserId) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  
  next();
}
// Middleware to ensure partner can only access their own data
async function requirePartnerScope(req, res, next) {
  const partnerUser = await db.partner_users.findByPk(req.session.partnerUserId);
  
  if (!partnerUser) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  
  // Attach partner context to request
  req.partnerContext = {
    userId: partnerUser.id,
    partnerId: partnerUser.partner_id,
    role: partnerUser.role
  };
  
  next();
}
